  r0.y = r0.x * v6.w + -PBAlphaTest.w;
  r0.x = v6.w * r0.x;
  r0.y = r0.y < 0.000000000e+000;
  if (r0.y != 0) discard;
  r0.yz = bumpMapSampler.Sample(bumpMapSampler_s, v1.xy).xy;
  r0.yz = r0.zy * float2(2.000000e+000,2.000000e+000) + float2(-1.000000e+000,-1.000000e+000);
  r0.yz = cb4[4].xx * r0.yz;
  r1.xy = v5.xy / v5.ww;
  r0.yz = r0.yz * r0.xx + r1.xy;
  o0.w = saturate(r0.x);

// For some unknown reason, this environment map sampler pulsates bright and
// dark from a distance in stereo, but I haven't managed to track down where
// this originates yet.
  r0.xyz = envMapSamplerRefr.Sample(envMapSamplerRefr_s, r0.yz).xyz;

// For now, kill the distortion when far away from the camera. That way we
// still get the effect up close where it matters, but side step the issue from
// a distance.
if (v5.w > 1)
	discard;

  o0.xyz = r0.xyz;
  return;
}
