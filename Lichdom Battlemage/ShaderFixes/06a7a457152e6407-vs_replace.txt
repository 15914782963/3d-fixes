void main(

{

  uint4 bitmask, uiDest;
  float4 fDest;

  r0.xyz = v0.xyz;
  r0.w = 1.000000000e+000;
  r1.x = dot(g_invGridMatrix._m10_m11_m12_m13, r0.xyzw);
  r1.xyzw = g_mCamera._m10_m11_m12_m13 * r1.xxxx;
  r2.x = dot(g_invGridMatrix._m00_m01_m02_m03, r0.xyzw);
  r0.x = dot(g_invGridMatrix._m20_m21_m22_m23, r0.xyzw);
  r1.xyzw = r2.xxxx * g_mCamera._m00_m01_m02_m03 + r1.xyzw;
  r0.xyzw = r0.xxxx * g_mCamera._m20_m21_m22_m23 + r1.xyzw;
  r0.xyzw = g_mCamera._m30_m31_m32_m33 + r0.xyzw;
  o0.xyzw = r0.xyzw;

// For accurate results on lighting where this shader is used, o2 needs to be
// derived from the original coordinates, o1 needs to be from adjusted
// coordinates. Copy this last chunk of code and perform it twice to find each
// output.

float4 orig_r0 = r0;
float4 orig_r1 = r1;
float4 orig_r2 = r2;

  orig_r0.xy = orig_r0.xy / orig_r0.ww;
  orig_r1.yz = orig_r0.xy * float2(5.000000e-001,5.000000e-001) + float2(5.000000e-001,5.000000e-001);
  orig_r1.xw = float2(1.000000e+000,1.000000e+000) + -orig_r1.zz;
  orig_r0.xyz = g_vFrustrumRB.xyz + -g_vFrustrumLB.xyz;
  orig_r0.xyz = orig_r1.yyy * orig_r0.xyz + g_vFrustrumLB.xyz;
  orig_r2.xyz = g_vFrustrumRT.xyz + -g_vFrustrumLT.xyz;
  orig_r1.yzw = orig_r1.yyy * orig_r2.xyz + g_vFrustrumLT.xyz;
  orig_r0.xyz = -orig_r1.yzw + orig_r0.xyz;
  orig_r0.xyz = orig_r1.xxx * orig_r0.xyz + orig_r1.yzw;
  orig_r1.x = dot(g_gridMatrix._m00_m01_m02, orig_r0.xyz);
  orig_r1.y = dot(g_gridMatrix._m10_m11_m12, orig_r0.xyz);
  orig_r1.z = dot(g_gridMatrix._m20_m21_m22, orig_r0.xyz);
  o2.xyz = orig_r1.xyz * orig_r0.www;

float4 stereo = StereoParams.Load(0);
r0.x += stereo.x * (r0.w - stereo.y);

  r0.xy = r0.xy / r0.ww;
  r1.yz = r0.xy * float2(5.000000e-001,5.000000e-001) + float2(5.000000e-001,5.000000e-001);
  r1.xw = float2(1.000000e+000,1.000000e+000) + -r1.zz;
  o1.xy = r1.yw * r0.ww;
  o1.z = r0.w;
  r0.xyz = g_vFrustrumRB.xyz + -g_vFrustrumLB.xyz;
  r0.xyz = r1.yyy * r0.xyz + g_vFrustrumLB.xyz;
  r2.xyz = g_vFrustrumRT.xyz + -g_vFrustrumLT.xyz;
  r1.yzw = r1.yyy * r2.xyz + g_vFrustrumLT.xyz;
  r0.xyz = -r1.yzw + r0.xyz;
  r0.xyz = r1.xxx * r0.xyz + r1.yzw;
  r1.x = dot(g_gridMatrix._m00_m01_m02, r0.xyz);
  r1.y = dot(g_gridMatrix._m10_m11_m12, r0.xyz);
  r1.z = dot(g_gridMatrix._m20_m21_m22, r0.xyz);
  o3.xyz = r0.xyz * r0.www;
  //o2.xyz = r1.xyz * r0.www;
  return;
}
